    def __init__(self, input_dim=2, hidden_dim=4, output_dim=1):
        # He åˆå§‹åŒ–
        self.W1 = np.random.randn(hidden_dim, input_dim) * np.sqrt(2/input_dim)
        self.b1 = np.zeros(hidden_dim)
        self.W2 = np.random.randn(output_dim, hidden_dim) * np.sqrt(2/hidden_dim)
        self.b2 = np.zeros(output_dim)

    def forward(self, x):

        self.x = x

        return self.y_pred

    def backward(self, x, y):

        dz2 = self.y_pred - y  # dL/dz2

        self.dW1 = np.outer(dz1, x)
        self.db1 = dz1

    def update(self, lr=0.01):

        self.b1 -= lr * self.db1
        self.W2 -= lr * self.dW2
        self.b2 -= lr * self.db2
